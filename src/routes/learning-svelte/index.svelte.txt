<script context="module" lang="ts">
  import type { Load } from '@sveltejs/kit';
  import { readdir } from 'fs-extra';

  export const load: Load = async ({}) => {
    const files = await readdir('src/routes/learning-svelte');
    return {
      props: {
        title: files.join(', ')
      }
    };
  };
</script>

<script lang="ts">
  export let title: string;
</script>

<main>
  {title}
</main>
